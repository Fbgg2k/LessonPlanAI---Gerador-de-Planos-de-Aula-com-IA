CREATE TABLE public.lesson_inputs (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  user_id uuid,
  tema text NOT NULL,
  nivel_ensino text NOT NULL,
  faixa_etaria text NOT NULL,
  duracao_prevista integer CHECK (duracao_prevista > 0),
  objetivo_especifico text NOT NULL,
  recursos_disponiveis text,
  criado_em timestamp with time zone DEFAULT timezone('utc'::text, now()),
  CONSTRAINT lesson_inputs_pkey PRIMARY KEY (id),
  CONSTRAINT lesson_inputs_user_id_fkey FOREIGN KEY (user_id) REFERENCES auth.users(id),
  CONSTRAINT fk_user FOREIGN KEY (user_id) REFERENCES auth.users(id)
);

  id uuid,

  tema text,
  nivel_ensino text,
  faixa_etaria text,

  objetivo_especifico text,
  recursos_disponiveis text,
  criado_em timestamp with time zone
);
CREATE TABLE public.lesson_plans (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  input_id uuid NOT NULL,
  introducao_ludica text NOT NULL,
  objetivo_bncc text NOT NULL,
  passo_a_passo text NOT NULL,
  rubrica_avaliacao text NOT NULL,
  resumo_geral text,
  criado_em timestamp with time zone DEFAULT timezone('utc'::text, now()),
  is_published boolean DEFAULT false,
  updated_at timestamp with time zone DEFAULT now(),
  conteudo text,
  grade_level text NOT NULL DEFAULT 'Not specified'::text,
  CONSTRAINT lesson_plans_pkey PRIMARY KEY (id),
  CONSTRAINT lesson_plans_input_id_fkey FOREIGN KEY (input_id) REFERENCES public.lesson_inputs(id),
  CONSTRAINT fk_lesson_input FOREIGN KEY (input_id) REFERENCES public.lesson_inputs(id)
);
CREATE TABLE public.profiles (
  id uuid NOT NULL,
  email text,
  name text,
  created_at timestamp with time zone DEFAULT now(),
  updated_at timestamp with time zone DEFAULT now(),
  CONSTRAINT profiles_pkey PRIMARY KEY (id),
  CONSTRAINT profiles_id_fkey FOREIGN KEY (id) REFERENCES auth.users(id)
);
CREATE TABLE public.users (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  name text NOT NULL,
  email text NOT NULL UNIQUE CHECK (email ~* '^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}$'::text),
  password text NOT NULL,
  created_at timestamp with time zone DEFAULT now(),
  updated_at timestamp with time zone DEFAULT now(),
  CONSTRAINT users_pkey PRIMARY KEY (id)
);
